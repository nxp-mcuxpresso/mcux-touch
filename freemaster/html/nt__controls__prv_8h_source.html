273 static inline int32_t _nt_control_overrun(struct nt_control_data *control)
274 {
275  if ((bool)_nt_control_get_flag(control, (int32_t)NT_CONTROL_NEW_DATA_FLAG))
276  {
277  return (int32_t)NT_FAILURE; /* overrun, data has not been processed */
278  }
279  _nt_control_clear_flag(control, (int32_t)NT_CONTROL_NEW_DATA_FLAG);
280  return (int32_t)NT_SUCCESS;
281 }
